📋 Guida per il nutrizionista

Per generare l'URL da fornire al paziente:

1Vai su script.google.com e crea un nuovo progetto
2Incolla lo script qui sotto nell'editor e salva
3Clicca Distribuisci → Nuova distribuzione → App Web
4Imposta accesso: "Chiunque" — copia il link e forniscilo al paziente

Script da copiare:

function writeData(data) { const ss = SpreadsheetApp.openById('IL_TUO_SPREADSHEET_ID'); const sheetName = data.tipo || 'Dati'; let sheet = ss.getSheetByName(sheetName); if (!sheet) sheet = ss.insertSheet(sheetName); if (sheet.getLastRow() === 0) { sheet.appendRow(Object.keys(data)); } sheet.appendRow(Object.values(data)); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); writeData(data); return ContentService.createTextOutput( JSON.stringify({status:'ok'}) ).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput( JSON.stringify({status:'error',msg:err.toString()}) ).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { if (e.parameter && e.parameter.payload) { const data = JSON.parse(e.parameter.payload); writeData(data); } return ContentService.createTextOutput('ok') .setMimeType(ContentService.MimeType.TEXT); } catch(err) { return ContentService.createTextOutput('error: '+err.toString()); } }

Sostituisci IL_TUO_SPREADSHEET_ID con l'ID del tuo Google Sheet (lo trovi nell'URL del foglio).
